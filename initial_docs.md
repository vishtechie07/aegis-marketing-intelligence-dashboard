## 1. PostgreSQL Database Schema
This schema uses a one-to-many relationship: one news article can trigger multiple AI-generated insights.
SQL
-- 1. Table for raw news harvested by the Java service
CREATE TABLE competitor_news (
    id SERIAL PRIMARY KEY,
    competitor_name VARCHAR(100) NOT NULL,
    title TEXT NOT NULL,
    content TEXT,
    source_url TEXT,
    published_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Table for AI-processed strategic insights (Ruflo Output)
CREATE TABLE agent_insights (
    id SERIAL PRIMARY KEY,
    news_id INTEGER REFERENCES competitor_news(id) ON DELETE CASCADE,
    agent_name VARCHAR(50), -- e.g., 'MarketAnalyst' or 'Strategist'
    threat_level INTEGER CHECK (threat_level BETWEEN 1 AND 10),
    summary TEXT,
    strategic_advice TEXT,
    processed_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- Index for fast lookup by competitor
CREATE INDEX idx_competitor_name ON competitor_news(competitor_name);

## 1. Vue 3 Composition API: The "Live Insight" Store
Using Pinia, you can manage the state of your insights across the entire application.
TypeScript
// src/stores/insightStore.ts
import { defineStore } from 'pinia';
import { ref } from 'vue';

export const useInsightStore = defineStore('insights', () => {
  const insights = ref<any[]>([]);

  // Function to add new insight with a 'new' flag for animation
  const addInsight = (newInsight: any) => {
    // Add to the beginning of the list
    insights.value.unshift({ ...newInsight, isNew: true });
    
    // Remove the 'isNew' flag after 3 seconds to stop the pulse
    setTimeout(() => {
      insights.value[0].isNew = false;
    }, 3000);

    // Keep only the last 50 items to prevent memory bloat
    if (insights.value.length > 50) insights.value.pop();
  };

  return { insights, addInsight };
});

## 2. The Real-Time Connector (EventSource)
Implement this in your main dashboard component to listen to the Java backend.
Code snippet
<script setup lang="ts">
import { onMounted, onUnmounted } from 'vue';
import { useInsightStore } from './stores/insightStore';

const store = useInsightStore();
let eventSource: EventSource | null = null;

onMounted(() => {
  // Connect to your Spring Boot SSE endpoint
  eventSource = new EventSource('http://localhost:8080/api/insights/stream');

  eventSource.onmessage = (event) => {
    const data = JSON.parse(event.data);
    store.addInsight(data);
  };

  eventSource.onerror = () => {
    console.error("SSE Connection Failed. Retrying...");
  };
});

onUnmounted(() => {
  if (eventSource) eventSource.close();
});
</script>
